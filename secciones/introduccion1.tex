\section{Definición}
\begin{frame}
\frametitle{Introducción}
\begin{block}

Sea \textcolor{blue}{$n$} un entero positivo. Una \textit{\textcolor{blue}{sucesión arbórea}} es una sucesión de enteros positivos \textcolor{blue}{ $d_1, d_2,\ldots d_n$} tal que \textcolor{blue}{ $\sum_{1\le i\le n} d_i=2(n-1)$}.
\end{block}
 
\begin{block}

Una sucesión \textcolor{blue}{ $\sigma= d_1, d_2,\ldots d_n$} es arbórea si y solo si hay un árbol cuyos vértices tienen grados \textcolor{blue}{ $d_1, d_2,\ldots d_n$}.
\end{block}
\end{frame}

\section{Objetivo de la plática}
\begin{frame}
\frametitle{Objetivo de la plática }
\begin{thm}
%\frametitle{Objetivo de la plática }
Sean \textcolor{blue}{$n\geq 4$} un número entero, \textcolor{blue}{$G$} una gráfica etiquetada con el conjunto de vértices \textcolor{blue}{ $V(G)= \left\lbrace w_1, w_2, \ldots, w_n \right\rbrace$} y \textcolor{blue}{$\sigma=d_1, d_2 \ldots, d_n$} una sucesión de grados arbórea, donde  \textcolor{blue}{$d_1 \leq d_2 \leq \cdots \leq d_n$}. El problema de decidir si \textcolor{blue}{$G$} tiene un árbol generador \textcolor{blue}{ $T$} tal que \textcolor{blue}{$d_T(w_i)=d_i$}, con \textcolor{blue} {$1 \leq i \leq n $}, es \textcolor{blue}{$NP$-completo}. 
\end{thm}

\begin{itemize}
 \Large
\centering \item Considerando a \textcolor{blue} {$B$} como el problema de decisión, debemos mostrar que \textcolor{blue}{$B$} esta en la clase \textcolor {blue} {$NP$}. 
\vspace{1.5 cm}
\centering \item  Mostrar que un problema de decisión \textcolor{blue}{A} que es \textcolor{blue}{NP-completo} es reducible en tiempo polinomial a \textcolor{blue}{$B$}. $$ \textcolor{blue}{A\leq _p B} $$

\end{itemize}
