\section{Objetivo de la plática}
\begin{frame}
\frametitle{Objetivo de la plática }
\begin{thm}
%\frametitle{Objetivo de la plática }
Sean \textcolor{blue}{$n\geq 4$} un número entero, \textcolor{blue}{$G$} una gráfica etiquetada con el conjunto de vértices \textcolor{blue}{ $V(G)= \left\lbrace w_1, w_2, \ldots, w_n \right\rbrace$} y \textcolor{blue}{$\sigma=d_1, d_2 \ldots, d_n$} una sucesión de grados arbórea, donde  \textcolor{blue}{$d_1 \leq d_2 \leq \cdots \leq d_n$}. El problema de decidir si \textcolor{blue}{$G$} tiene un árbol generador \textcolor{blue}{ $T$} tal que \textcolor{blue}{$d_T(w_i)=d_i$}, con \textcolor{blue} {$1 \leq i \leq n $}, es \textcolor{blue}{$NP$-completo}. 

\end{thm}
\end{frame}
